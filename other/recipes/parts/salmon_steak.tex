\title{Стейк из лосося}
\NbPerson{2}
%------------------------------------------

\begin{ingredient}
%\vspace{0.5cm}
\begin{main}
	\item Лосось, 0.3\,кг
	\item Лимон, 0.5\,шт.
	\item Оливковое масло, 3\,ст.л.
	\item Соевый соус, 3\,ст.л.
\end{main}
\end{ingredient}
\begin{recipe}
\setcounter{step}{0}
\step{Подготовить мясо: удалить косточки, чешую.}
\step{Выдавить половинку лимона, поперчить, смешать с оливковым мыслом и соусом, замариновать рыбку минут хотя бы на 20.}
\step{Перед жаркой промокнуть с рыбы лишний маринад.}
\step{Пожарить, лучше на гриле, около 5 мин.}


\end{recipe}

\begin{notes}
	Очень хорошо подходит к круглозерному рису.
\end{notes}
