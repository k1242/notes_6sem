\title{Соба с грибочками в сливочном соусе}
\NbPerson{2}
%------------------------------------------

\begin{ingredient}
%\vspace{0.5cm}
\begin{main}
	\item Соба
	\item Сливки 10-20\%, 100\,мл
	\item Пармезан, чуть-чуть
	\item Шампиньоны
	\item Желток, 2\,шт.
	\item \sout{\text{Бекон, 0.25\,кг}}
\end{main}
\end{ingredient}
\begin{recipe}
\step{Пожарить грибочки на сильном огне.}
\step{Смешать два желтка, сливки и мелко натертый сыр.}
\step{Сварить собу, лучше сразу после варки разложить порции и залить сливочным соусом, добавить грибочки.}


\end{recipe}

\begin{notes}
	Можно ещё пожарить и добавить бекон, тогда получится карбонара на собе с грибочками.
\end{notes}
